108 void ControlsPanel::addResourceFolderSelected() {
109  std::string resourceFolderPath = FilesystemResourceProvider::getResourceFolderPath(this->folderDialog.GetSelected().string());
110  if (resourceFolderPath.empty())
111  return Device::popupError(TR("error.modelviewer.resource_folder_not_valid"));
112 
113  bool resourceExists = false;
114  for (const auto& fileProvider : Resource::getResourceProviders(FILESYSTEM_PROVIDER_NAME)) {
115  if (resourceFolderPath == assert_cast<FilesystemResourceProvider*>(fileProvider.get())->getFolder()) {
116  resourceExists = true;
117  break;
118  }
119  }
120  if (!resourceExists)
121  Resource::addResourceProvider(new FilesystemResourceProvider{resourceFolderPath});
122  else
123  Device::popupError(TR("error.modelviewer.resource_folder_already_registered"));
124 }
