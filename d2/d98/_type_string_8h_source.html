21 struct TypeStringBase {
22  std::string name, prefix, suffix;
23  explicit TypeStringBase(const std::string& name_) {
24  this->name = name_;
25  }
26  std::string operator()() const {
27  std::ostringstream os;
28  if (!this->prefix.empty()) os << this->prefix << ' ';
29  os << type();
30  if (!this->suffix.empty()) os << ' ' << this->suffix;
31  if (!this->name.empty()) os << ' ' << this->name;
32  return os.str();
33  }
34  [[nodiscard]] virtual std::string type() const = 0;
35  virtual TypeStringBase& as_param() {
36  return *this;
37  }
38 };

40 template<typename T>
41 struct TypeString : TypeStringBase {
42  explicit TypeString(const std::string& name_ = "") : TypeStringBase(name_) {}
43  [[nodiscard]] std::string type() const override {
44  std::string type = typeid(T).name();
45 #ifdef CHIRA_AS_USE_DEMANGLED_NAMES
46  int status;
47  char* demangled_name = abi::__cxa_demangle(type.c_str(), nullptr, nullptr, &status);
48  if (status == 0) {
49  type = demangled_name;
50  std::free(demangled_name);
51  }
52 #endif
53  return type;
54  }
55 };
