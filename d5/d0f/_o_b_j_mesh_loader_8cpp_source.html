85 void OBJMeshLoader::addVertex(Vertex v, Index* currentIndex, std::vector<Vertex>& vertices, std::vector<Index>& indices) {
86  auto position = std::find(vertices.begin(), vertices.end(), v);
87  if (position != vertices.end()) {
88  indices.push_back(static_cast<Index>(position - vertices.begin()));

105  }
106  for (Index i = 1; i < indices.size() + 1; i += 3) {
107  meshDataStream << "f " << i << '/' << i << '/' << i << ' '
108  << i+1 << '/' << i+1 << '/' << i+1 << ' '
109  << i+2 << '/' << i+2 << '/' << i+2 << '\n';
110  }
111  meshDataStream >> std::noskipws;
112 
113  std::vector<byte> out;
